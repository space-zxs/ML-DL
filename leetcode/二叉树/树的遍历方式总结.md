        dfs(root);
        return res;
        
    }  
private:
    vector<int> res;
    void dfs(TreeNode* root) {
        if (!root) return;
        dfs(root->left);
        res.push_back(root->val);
        dfs(root->right);
    }

### 非递归实现

```
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    vector<int> inorderTraversal(TreeNode* root) {
        stack<TreeNode*> st;
        vector<int> res;
        if(!root) return res;
        while(!st.empty() || root!=nullptr){
            while(root){
                st.push(root);
                root = root -> left;
            }
            root = st.top();
            st.pop();
            res.push_back(root->val);
            root = root->right;
        }
        return res;
    }
};
```

### 非递归实现
### 思路
  - 类似于中序遍历，只不过值得输出位置有变化
  - 遍历树的根节点的时候就把值入栈
  - 出栈，遍历右节点
```
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    vector<int> preorderTraversal(TreeNode* root) {
        stack<TreeNode*> st;
        vector<int> res;
        TreeNode* node = root;
        while(!st.empty() || node){
            while(node){
                res.push_back(node->val);
                st.push(node);
                node = node->left;
            }
            node = st.top();
            st.pop();
            node = node -> right;
        }
        return res;
    }
};

### 递归实现

### 思路：
   - 设置一个辅助栈模拟递归的过程。
   - 与前序和中序的变化方式有所不同
   - 需要在遍历完右子树之后
   - 判断右子树是否遍历完是关键，并且还要，避免重复加入右子树值的情况
   - 如果右子树为空，或者 右子树已经遍历过，那么可以往数组入值
   - 否则遍历右子树，记录上一个遍历的结点。
```
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    vector<int> postorderTraversal(TreeNode* root) {
        stack<TreeNode*> st;
        vector<int> res;
        if(!root) return res;
        TreeNode* node = root;
        TreeNode* prev = nullptr; // 记录上一个访问结点
        while(!st.empty() || node){
            while(node){
                st.push(node);
                node = node -> left;
            }
            node = st.top();
            st.pop();
            // 如果右子树存在并且没访问过
            if(node-> right && prev != node->right){
                st.push(node);// 左结点重新push回去
                node = node -> right;//访问右节点
            }else{
                //填充vec在node->left和node->right后面
                //此时左右子树都访问完成
                res.push_back(node->val);
                prev = node;
                node = nullptr;//避免重复访问右子树
            }
        }
        return res;
    }
};
